<script>
function togglePrintType() {
    const printType = document.getElementById("printType").value;
    document.getElementById("countSection").style.display = printType === "count" ? "block" : "none";
    document.getElementById("rangeSection").style.display = printType === "range" ? "block" : "none";
}

function createTicket(number, prefix, organizationName) {
    let now = new Date();
    let date = now.toLocaleDateString("ar-EG");
    let time = now.toLocaleTimeString("ar-EG");

    let ticket = document.createElement("div");
    ticket.className = "ticket";

    ticket.innerHTML = `
        <div class="org-name">${organizationName}</div>
        <div class="date-time">${date} - ${time}</div>
        <div class="number">${prefix}${String(number).padStart(3, '0')}</div>
        <div class="footer">يرجى الانتظار حتى يتم مناداة رقمك</div>
    `;

    return ticket;
}

function printNumbers() {
    let printType = document.getElementById("printType").value;
    let prefix = document.getElementById("prefix").value || "";
    let organizationName = document.getElementById("organizationName").value || "جمعية الاحسان الخيرية";
    let printArea = document.getElementById("printArea");

    printArea.innerHTML = "";

    if (printType === "count") {
        let count = parseInt(document.getElementById("count").value);
        let mode = document.getElementById("mode").value;

        if (!count || count <= 0) {
            alert("أدخل عدد صحيح");
            return;
        }

        if (mode === "reset") {
            localStorage.setItem("lastNumber", 0);
        }

        let lastNumber = parseInt(localStorage.getItem("lastNumber")) || 0;

        for (let i = 1; i <= count; i++) {
            lastNumber++;
            printArea.appendChild(createTicket(lastNumber, prefix, organizationName));
        }

        localStorage.setItem("lastNumber", lastNumber);
    }

    if (printType === "range") {
        let fromNumber = parseInt(document.getElementById("fromNumber").value);
        let toNumber = parseInt(document.getElementById("toNumber").value);

        if (!fromNumber || !toNumber || fromNumber <= 0 || toNumber <= 0) {
            alert("أدخل أرقام صحيحة");
            return;
        }

        if (fromNumber > toNumber) {
            alert("رقم البداية يجب أن يكون أصغر من أو يساوي رقم النهاية");
            return;
        }

        for (let number = fromNumber; number <= toNumber; number++) {
            printArea.appendChild(createTicket(number, prefix, organizationName));
        }

        localStorage.setItem("lastNumber", toNumber);
    }

    window.print();
    printArea.innerHTML = "";
}

function resetCounter() {
    if (confirm("هل أنت متأكد من تصفير العداد؟")) {
        localStorage.setItem("lastNumber", 0);
        alert("تم تصفير العداد");
    }
}
